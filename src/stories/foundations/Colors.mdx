# Colors

All colors are defined as CSS custom properties in `src/tokens/colors.css`.

export const Swatch = ({ color, label, token }) => (
  <div style={{ display: 'flex', alignItems: 'center', gap: 12, marginBottom: 8 }}>
    <div style={{ width: 48, height: 48, borderRadius: 8, background: color, border: '1px solid rgba(0,0,0,0.08)', flexShrink: 0 }} />
    <div>
      <div style={{ fontSize: 13, fontWeight: 600, color: '#141414' }}>{label}</div>
      <code style={{ fontSize: 11, color: '#ADADAD' }}>{token}</code>
    </div>
  </div>
)
