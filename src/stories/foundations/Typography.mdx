# Typography

All type tokens are defined in `src/tokens/typography.css`. The base font is **Inter** (300–700 weight, Cyrillic + Latin).

## Font Sizes

export const sizes = [
  { token: '--font-size-xs',    value: '12px', label: 'XS — badges, hints, transfer labels' },
  { token: '--font-size-sm',    value: '14px', label: 'SM — meta, captions' },
  { token: '--font-size-label', value: '14px', label: 'Label — form labels, section headings' },
  { token: '--font-size-md',    value: '16px', label: 'UI — default UI text, prominent labels' },
  { token: '--font-size-lg',    value: '16px', label: 'Body — body text' },
  { token: '--font-size-title', value: '18px', label: 'Title — card headings, counters' },
]

<div style={{ display: 'flex', flexDirection: 'column', gap: 16, fontFamily: 'Inter, system-ui, sans-serif', padding: '8px 0' }}>
  {sizes.map(s => (
    <div key={s.token} style={{ display: 'flex', alignItems: 'baseline', gap: 24 }}>
      <span style={{ fontSize: s.value, color: '#141414', minWidth: 220 }}>
        {s.label}
      </span>
      <code style={{ fontSize: 12, color: '#ADADAD' }}>{s.token} · {s.value}</code>
    </div>
  ))}
</div>

## Font Weights

<div style={{ display: 'flex', gap: 32, fontFamily: 'Inter, system-ui, sans-serif', padding: '8px 0', flexWrap: 'wrap' }}>
  {[
    { token: '--font-weight-light',   value: 300, label: 'Light' },
    { token: '--font-weight-regular', value: 400, label: 'Regular' },
    { token: '--font-weight-medium',  value: 500, label: 'Medium' },
    { token: '--font-weight-bold',    value: 700, label: 'Bold' },
  ].map(w => (
    <div key={w.token} style={{ textAlign: 'center' }}>
      <div style={{ fontWeight: w.value, fontSize: 24, color: '#141414' }}>{w.label}</div>
      <code style={{ fontSize: 11, color: '#ADADAD' }}>{w.value}</code>
    </div>
  ))}
</div>
